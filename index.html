1."We are interested in retrieving only the students' names; all the names should be in caps. ['JOHN', 'BABA', 'YAGA', 'WICK']" 

function retrievingNames(){
    let studentRecords = [ {name: 'John', id: 123, marks : 98 }, 
    {name:'Baba', id: 101, marks : 23 },
    {name: 'yaga', id: 200, marks : 45 },
    {name: 'Wick', id: 115, marks : 75 } ]
    let res=[];
    for(let i=0;i<studentRecords.length;i++){
        let obj=studentRecords[i];
        res.push(obj.name.toLocaleUpperCase());
    }
    console.log(res);
}
retrievingNames();

    {name:'Baba', id: 101, marks : 23 },

    {name: 'Wick', id: 115, marks : 75 } ]
    let sum=0;
    for(let i=0;i<studentRecords.length;i++){
        let obj=studentRecords[i];
        if(obj.marks<50){
            obj.marks=obj.marks+15;
            if(obj.marks>50){
            sum+=obj.marks;
        }  
        }
    }
    console.log(sum);
}
